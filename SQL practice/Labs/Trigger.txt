--Расход бензина за рейс не должен превышать 100 литров.

CREATE TRIGGER FuelForTrip
ON [Trip] AFTER INSERT AS
	IF EXISTS (
		SELECT ID
		FROM [Car] INNER JOIN [Trip] ON ID = CarID
		WHERE (FuelConsumption * Distance / 100) > 100)
	BEGIN
		ROLLBACK
	END


--Проверка
